Each panel is represented by a tab in its group's header. Dockview ships a default tab renderer, and this page covers replacing or extending it: custom tab components, accessing panel parameters, context menus, full-width and reorder behaviour, and tab groups.

## Register a tab component

<FrameworkSpecific framework='React'>
```tsx
const components = {
  tab_1: (props: IDockviewPanelHeaderProps) => {
    const api: DockviewPanelApi  = props.api;
    const containerApi: DockviewApi  = props.containerApi;

    return <div>{/** logic */}</div>

},
tab_2: (props: IDockviewPanelHeaderProps) => {
return <div>{/* * logic */}</div>
}
};

return <DockviewReact tabComponents={components}/>

````
</FrameworkSpecific>

<FrameworkSpecific framework='Vue'>
```ts
const App = {
    name: 'App',
    components: {
        'component_1': VueComponent1,
        'tab_1': VueComponent2,
    },
    methods: {
        onReady(event: DockviewReadyEvent) {
            event.api.addPanel({
                id: 'panel_1',
                component: 'component_1',
                tabComponent: 'tab_1'
            });
        },
    },
    template: `
      <dockview-vue
        @ready="onReady"
      >
      </dockview-vue>`,
};
````

</FrameworkSpecific>

<FrameworkSpecific framework='Angular'>
```ts
@Component({
    selector: 'tab-component-1',
    template: `<div>Custom Tab 1</div>`
})
export class TabComponent1 {
    @Input() api: DockviewPanelApi;
    @Input() containerApi: DockviewApi;
}

@Component({
selector: 'tab-component-2',
template: `<div>Custom Tab 2</div>`
})
export class TabComponent2 {
@Input() api: DockviewPanelApi;
@Input() containerApi: DockviewApi;
}

@Component({
selector: 'app-root',
template: `         <dv-dockview 
            [tabComponents]="tabComponents"
            (ready)="onReady($event)">
        </dv-dockview>
    `
})
export class AppComponent {
tabComponents = {
'tab_1': TabComponent1,
'tab_2': TabComponent2,
};

    onReady(event: DockviewReadyEvent) {
        event.api.addPanel({
            id: 'panel_1',
            component: 'component_1',
            tabComponent: 'tab_1'
        });
    }

}

````
</FrameworkSpecific>

<FrameworkSpecific framework='JavaScript'>
```ts
class CustomTabRenderer implements IHeaderRenderer {
    private readonly _element: HTMLElement;

    get element(): HTMLElement {
        return this._element;
    }

    constructor() {
        this._element = document.createElement('div');
    }

    init(parameters: HeaderPartInitParameters): void {
        const api: DockviewPanelApi = parameters.api;
        const containerApi: DockviewApi = parameters.containerApi;

        this._element.textContent = api.title || 'Custom Tab';
    }

    dispose(): void {
        // Cleanup logic
    }
}

const api = createDockview(parentElement, {
    createTabComponent: (options) => {
        switch (options.name) {
            case 'tab_1':
                return new CustomTabRenderer();
            default:
                return undefined;
        }
    },
});

api.addPanel({
    id: 'panel_1',
    component: 'component_1',
    tabComponent: 'tab_1'
});
````

</FrameworkSpecific>

class DefaultTabRenderer implements IHeaderRenderer {
    private readonly _element: HTMLElement;

    get element(): HTMLElement {
        return this._element;
    }

    constructor() {
        this._element = document.createElement('div');
        this._element.className = 'custom-default-tab';
    }

    init(parameters: HeaderPartInitParameters): void {
        const api: DockviewPanelApi = parameters.api;
        const containerApi: DockviewApi = parameters.containerApi;

        this._element.textContent = api.title || 'Default Tab';
    }

    dispose(): void {
        // Cleanup logic
    }
}

const api = createDockview(parentElement, {
    createTabComponent: (options) => {
        // Return the default tab renderer for all tabs
        return new DefaultTabRenderer();
    },
});

## Accessing custom panel parameters

You can provide a generic type that matches the structure of the expected custom panels parameters
to provide type-hints for the panel parameters which can be accessed via the `params` option.

<FrameworkSpecific framework='React'>
```jsx
type MyParameters = { my_value: number };

const MyTab = (props: IDockviewPanelHeaderProps<MyParameters>) => {
const value: number = props.params.my_value;
return <div>{/* * logic */}</div>
}

````
</FrameworkSpecific>

<FrameworkSpecific framework='Vue'>
```ts
// In your Vue tab component
import { defineComponent } from 'vue';
import type { IDockviewPanelHeaderProps } from 'dockview-vue';

type MyParameters = { my_value: number };

export default defineComponent({
    props: {
        params: {
            type: Object as () => MyParameters,
            required: true,
        },
    },
    template: `<div>{{ params.my_value }}</div>`,
});
````

</FrameworkSpecific>

<FrameworkSpecific framework='Angular'>
```ts
import { Component, Input } from '@angular/core';
import type { IDockviewPanelHeaderProps } from 'dockview-angular';

type MyParameters = { my_value: number };

@Component({
selector: 'my-tab',
template: `<div>{{ params.my_value }}</div>`,
})
export class MyTab {
@Input() params: MyParameters;
}

````
</FrameworkSpecific>

<FrameworkSpecific framework='JavaScript'>
```ts
import type { HeaderPartInitParameters } from 'dockview';

type MyParameters = { my_value: number };

class MyTab implements IHeaderRenderer {
    private readonly _element: HTMLElement;

    get element(): HTMLElement {
        return this._element;
    }

    constructor() {
        this._element = document.createElement('div');
    }

    init(parameters: HeaderPartInitParameters): void {
        // parameters.params is typed as unknown; cast to your type
        const params = parameters.params as MyParameters;
        const value: number = params.my_value;
        this._element.textContent = String(value);
    }

    dispose(): void {}
}
````

</FrameworkSpecific>

## Extend the default tab implementation

If you only want to make minor changes to the tab rendering you may be able to use the
default implementation as a base. This could include:

- Hiding the close button
- Attaching additional event listeners

<FrameworkSpecific framework='React'>
```tsx
import { IDockviewPanelHeaderProps, DockviewDefaultTab } from 'dockview-react';

const MyCustomTab = (props: IDockviewPanelHeaderProps) => {
const onContextMenu = (event: React.MouseEvent) => {
event.preventDefault();
console.log('context menu', props.api.id);
};
return <DockviewDefaultTab onContextMenu={onContextMenu} hideClose={true} {...props} />;
};

````
</FrameworkSpecific>

<FrameworkSpecific framework='Vue'>
:::info
`DockviewDefaultTab` is a React component. In Vue, implement tab extensions by composing your own tab component from scratch using the tab API and event listeners.
:::
</FrameworkSpecific>

<FrameworkSpecific framework='Angular'>
:::info
`DockviewDefaultTab` is a React component. In Angular, implement tab extensions by composing your own tab component from scratch using the injected `DockviewPanelApi`.
:::
</FrameworkSpecific>

<FrameworkSpecific framework='JavaScript'>
:::info
`DockviewDefaultTab` is a React component. In JavaScript, implement tab extensions by subclassing or composing your own `IHeaderRenderer` from scratch.
:::
</FrameworkSpecific>

## Tab context menu

Right-clicking a tab can show a context menu. This is opt-in; no menu is shown unless
`getTabContextMenuItems` is provided. Return an empty array to suppress the menu for specific panels.

### Built-in items

Pass string shortcuts to render standard menu entries without any extra code:

| Value | Behaviour |
| --- | --- |
| `'close'` | Close the right-clicked panel |
| `'closeOthers'` | Close every other panel in the same group |
| `'closeAll'` | Close all panels in the group |
| `'closeLeft'` | Close the panels to the left of the right-clicked panel (disabled when it is the first tab) |
| `'closeRight'` | Close the panels to the right of the right-clicked panel (disabled when it is the last tab) |
| `'maximize'` | Maximize the group. Renders as `Restore` when the group is already maximized, and is disabled for floating / popout panels that cannot be maximized |
| `'float'` | Move the panel into a floating window (disabled when it is already floating) |
| `'popout'` | Move the panel into a new browser window (disabled when it is already popped out) |
| `'separator'` | Render a visual divider |

### Custom label items

Provide an object with a `label` and `action` to add a simple clickable entry:

```ts
getTabContextMenuItems: (params) => [
    { label: 'Log panel id', action: () => console.log(params.panel.id) },
    { label: 'Disabled entry', action: () => {}, disabled: true },
]
````

### Custom component items

For richer UI, provide a framework component via the `component` field.
The component receives `panel`, `group`, `api`, `close`, and an optional `componentProps` object.

<FrameworkSpecific framework="React">
```tsx
import { IContextMenuItemComponentProps } from 'dockview-react';

const FloatMenuItem = ({ panel, api, close }: IContextMenuItemComponentProps) => (

<div
    className="dv-context-menu-item"
    onClick={() => {
        api.addFloatingGroup(panel);
        close();
    }}
>
    Float tab
</div>
);

<DockviewReact
    getTabContextMenuItems={() => [
        'close',
        'separator',
        { component: FloatMenuItem },
    ]}
/>
```
</FrameworkSpecific>

<FrameworkSpecific framework="Vue">
```ts
// Vue component receives props under a `params` wrapper: params.panel, params.api, etc.
const FloatMenuItem = defineComponent({
    props: { params: { type: Object as PropType<IContextMenuItemComponentProps>, required: true } },
    template: `
        <div class="dv-context-menu-item" @click="onClick">Float tab</div>
    `,
    methods: {
        onClick() {
            this.params.api.addFloatingGroup(this.params.panel);
            this.params.close();
        },
    },
});
```
</FrameworkSpecific>

<FrameworkSpecific framework="Angular">
```ts
// Angular component receives panel, group, api, close, componentProps as @Input() bindings.
@Component({
    selector: 'float-menu-item',
    template: `<div class="dv-context-menu-item" (click)="onClick()">Float tab</div>`,
})
export class FloatMenuItemComponent {
    @Input() panel: IDockviewPanel;
    @Input() api: DockviewApi;
    @Input() close: () => void;

    onClick() {
        this.api.addFloatingGroup(this.panel);
        this.close();
    }

}

````
</FrameworkSpecific>

<FrameworkSpecific framework="JavaScript">
```ts
// Implement IContextMenuItemRenderer to render a custom DOM element.
class FloatMenuItem {
    readonly element = document.createElement('div');

    init({ panel, api, close }) {
        this.element.className = 'dv-context-menu-item';
        this.element.textContent = 'Float tab';
        this.element.addEventListener('click', () => {
            api.addFloatingGroup(panel);
            close();
        });
    }

    dispose() {}
}
````

</FrameworkSpecific>

## Tab reorder mode

Controls how tabs animate when reordered via drag and drop.

| Mode        | Behaviour                                                                       |
| ----------- | ------------------------------------------------------------------------------- |
| `'default'` | Tabs swap positions instantly on drop; the original behaviour.                 |
| `'smooth'`  | Tabs slide smoothly to open a gap for the dragged tab, similar to browser tabs. |

<CodeRunner id="dockview/tabview" />

<DocRef declaration="DockviewTheme" methods={['tabAnimation']} />

<FrameworkSpecific framework="React">
```tsx
import { themeAbyss } from 'dockview-react';

const myTheme = { ...themeAbyss, tabAnimation: 'smooth' };

return <DockviewReact theme={myTheme} />

````
</FrameworkSpecific>

<FrameworkSpecific framework='Vue'>
```vue
<script setup lang="ts">
import { themeAbyss } from 'dockview-vue';

const theme = { ...themeAbyss, tabAnimation: 'smooth' as const };
</script>

<template>
    <dockview-vue :theme="theme" @ready="onReady" />
</template>
````

</FrameworkSpecific>

<FrameworkSpecific framework='Angular'>
```ts

@Component({
template: `         <dv-dockview
            [theme]="theme"
            (ready)="onReady($event)">
        </dv-dockview>
    `,
})

// ...
}

````
</FrameworkSpecific>

<FrameworkSpecific framework='JavaScript'>
```ts

    theme: { ...themeAbyss, tabAnimation: 'smooth' },

````

## Tab groups

Tab groups let you visually organise tabs within a group using coloured chips,
with a custom colour palette, chip renderers, context menus, and serialization.

See [Tab groups](/docs/core/panels/tabGroups) for the full guide.

## Tab height

Tab height can be controlled through CSS.

```css
:root {
    --dv-tabs-and-actions-container-height: 35px;
}
```
